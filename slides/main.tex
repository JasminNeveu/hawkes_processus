\begin{frame}{Processus de comptage et processus ponctuels}
  \begin{definition}[Processus de comptage]
    Un processus de comptage $N(t)_{t \geq 0}$ est un processus stochastique partant de $N_0 = 0$, dont les trajectoires sont croissantes par sauts d'amplitude 1, continues à droite.
  \end{definition}

  \pause
  \begin{definition}[Processus ponctuel]
    Un processus ponctuel $\mathbf{T} = \{T_1,T_2,...\}$ est une suite de variables aléatoires dans $[0, \inf)$ telle que $\mathbb{P}(0 \leq T_1 \leq T_2  \leq ...) = 1$ et le nombre de points dans une région bornée est fini p.s
  \end{definition}
\end{frame}

\begin{frame}{Fonction de densité et fonction de répartition}
  
  \blue{Filtration naturelle}:

  \begin{equation*}\mathcal{H}(u) = \sigma\bigl(T_n, n \leq u\bigr)\end{equation*}
  \pause
  \blue{Fonction de répartition}:
  \begin{equation*}
    \begin{aligned}
      F^*\bigl(t \mid \mathcal{H}(u)\bigr) &= \int_u^t \mathbb{P} \bigl(T_{k+1} \in [s,s+\mathrm{d}s] \bigm| \mathcal{H}(u)\bigr) \, \mathrm{d}s \\
                                           &= \int_u^t f^*(s \mid \mathcal{H}(u)) \, \mathrm{d}s.
  \end{aligned}
  \end{equation*}

  \pause
  \blue{Fonction de densité}:
  \begin{equation*}
  f(t_1,t_2,...,t_k) = \prod_{i=1}^k f^*(t_i | \mathcal{H}(t_{i-1})).
  \end{equation*}

\end{frame}

\begin{frame}{Intensité conditionnelle}
  Soit $(N_t)_{t \geq 0})$ un processus de comptage.
\begin{definition}
\begin{equation*}
\lambda^*(t) = \lim_{h \to 0^{+}} \frac{\mathbb{E}\bigl[N(t+h)-N(t) | \mathcal{H}(t)\bigr]}{h}
\end{equation*}
\end{definition}

\pause
\blue{Auto-exitation} \\

Si une arrivée cause une augmentation de l'intensité conditonnelle, alors on dit que le processus est auto-exitant.

\pause

\blue{Auto-régulé}\\
Si la fonction d'intensité conditionnelle baisse après une arrivée alors le processus est dit auto-régulé.

\pause

\blue{Explosion}  \\

$N(t) - N(s) = \inf, ~\text{pour}~ t - s  \leq \inf$


\end{frame}

\begin{frame}{Compensateur}
  \begin{definition}[Compensateur]
    Soit $(N_t)_{t \geq 0}$  un processus de comptage croissant, la fonction 
    \begin{equation*}
      \Lambda(t) = \int_{0}^{t}\lambda^*(s) ds
    \end{equation*}
    est appelée compensateur du processus.
  \end{definition}

  \pause 
  C'est l'unique fonction fonction prédictible issue de 0 telle que 
  \begin{equation*}
    N(t) = M(t) + \Lambda(t)
  \end{equation*}
  avec $M(t)$ une martingale

\end{frame}
 

\section{Processus de Hawkes}

\begin{frame}
  \begin{definition}[Processus de Hawkes]
    Soit $(N_t)_{t\geq 0}$ un processus de comptage, tel que 
    \begin{equation*}
      \mathbb{P}\bigl(N(t+h) - N(t) = m | \mathcal{H}(t)\bigr) = \begin{cases}
        \lambda^*(t)h + o(h), &m=1 \\
        o(h), &m > 1\\
        1 - \lambda^*(t)h + o(h), &m=0
      \end{cases}
    \end{equation*}
  \end{definition}
 \pause 
 et tel que la fonction d'intensité peut s'écrire:
  \begin{equation*}
    \lambda^*(t) = \lambda + \sum_{t_i \leq t} \mu(t-t_i)
  \end{equation*}
  avec $lambda > 0$ et $\mu :(0, \inf) \rightarrow [0,\inf)$ respectivement \textit{intensité de fond} et \textit{fonction d'exitation}

  \pause
  \begin{remark}
    Si $\mu(\text{·}) = 0 $, le processus se réduit à un processus de Poisson homogène.
  \end{remark}

\end{frame}

\begin{frame}{Fonction d'exitation}
  Généralement on définit la fonction d'exitation de la manière suivante:
  \begin{equation*}
    \mu(t) = \alpha \exp^{-\beta t}
  \end{equation*}
\end{frame}
